Go——排序

1. 使用 sort.Interface 接口进行排序

sort.Interface 接口的定义：

type Interface interface {
	// Len 集合中元素的数量
	Len() int

	// Less 索引为 i 的元素是否必须排在索引为 j 的元素之前
	Less(i, j int) bool

	// Swap 交换索引为 i 和 j 的元素
	Swap(i, j int)
}

示例：

type MyStringList []string

func (m MyStringList) Len() int {
	return len(m)
}

func (m MyStringList) Less(i, j int) bool {
	return m[i] < m[j]
}

func (m MyStringList) Swap(i, j int) {
	m[i], m[j] = m[j], m[i]
}

func main() {

	// 准备一个内容被打乱顺序的字符串切片
	names := MyStringList{
		"zhangsan",
		"lisi",
		"wamgwu",
		"zhaoliu",
	}

	// 使用sort包进行排序
	sort.Sort(names)

	// 遍历打印结果
	for _, v := range names {
		fmt.Printf("%s\n", v)
	}
	/* 执行结果：
	lisi
	wamgwu
	zhangsan
	zhaoliu
	*/

	sort.Sort(sort.Reverse(names))

	for _, v := range names {
		fmt.Printf("%s\n", v)
	}
	/* 执行结果：
	zhaoliu
	zhangsan
	wamgwu
	lisi
	*/

}

3. 对结构体数据进行排序

3.1 实现 sort.Interface 进行结构体排序

// HeroKind 声明英雄的分类
type HeroKind int

// 定义HeroKind常量, 类似于枚举
const (
	None HeroKind = iota
	Tank
	Assassin
	Mage
)

type Hero struct {
	Name string  // 英雄的名字
	Kind HeroKind  // 英雄的种类
}


type Heroes []*Hero


func (s Heroes) Len() int {
	return len(s)
}


func (s Heroes) Less(i, j int) bool {

	// 如果英雄的分类不一致时, 优先对分类进行排序
	if s[i].Kind != s[j].Kind {
		return s[i].Kind < s[j].Kind
	}

	// 默认按英雄名字字符升序排列
	return s[i].Name < s[j].Name
}

func (s Heroes) Swap(i, j int) {
	s[i], s[j] = s[j], s[i]
}

func main() {

	// 准备英雄列表
	heroes := Heroes{
		&Hero{"吕布", Tank},
		&Hero{"李白", Assassin},
		&Hero{"妲己", Mage},
		&Hero{"貂蝉", Assassin},
		&Hero{"关羽", Tank},
		&Hero{"诸葛亮", Mage},
	}

	// 使用sort包进行排序
	sort.Sort(heroes)

	// 遍历英雄列表打印排序结果
	for _, v := range heroes {
		fmt.Printf("%+v\n", v)
	}
}

3.2 使用 sort.Slice 进行切片元素排序

从 Go 1.8 开始，Go语言在 sort 包中提供了 sort.Slice() 函数进行更为简便的排序方法。sort.Slice() 函数只要求传入需要排序的数据，以及一个排序时对元素的回调函数，类型为 func(i,j int)bool，sort.Slice() 函数的定义如下：

func Slice(x interface{}, less func(i, j int) bool) {
	rv := reflectValueOf(x)
	swap := reflectSwapper(x)
	length := rv.Len()
	quickSort_func(lessSwap{less, swap}, 0, length, maxDepth(length))
}

使用示例：

type HeroKind int

const (
	None = iota
	Tank
	Assassin
	Mage
)

type Hero struct {
	Name string
	Kind HeroKind
}

func main() {

	heroes := []*Hero{
		{"吕布", Tank},
		{"李白", Assassin},
		{"妲己", Mage},
		{"貂蝉", Assassin},
		{"关羽", Tank},
		{"诸葛亮", Mage},
	}

	sort.Slice(heroes, func(i, j int) bool {
		if heroes[i].Kind != heroes[j].Kind {
			return heroes[i].Kind < heroes[j].Kind
		}

		return heroes[i].Name < heroes[j].Name
	})

	for _, v := range heroes {
		fmt.Printf("%+v\n", v)
	}
}
