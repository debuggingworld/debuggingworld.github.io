Go——文件操作

1. 目录操作

// 创建目录
func createDir() {
	// 创建单个目录
	// err := os.Mkdir("test", os.ModePerm)
	err := os.MkdirAll("test/a/b", os.ModePerm)
	if err != nil {
		fmt.Printf("err: %v\n", err)
	}
}

// 删除目录
func removeDir() {
	// err := os.Remove("test.txt")

	err := os.RemoveAll("test")
	if err != nil {
		fmt.Printf("err: %v\n", err)
	}
}

// 获得工作目录
func getWd() {
	dir, err := os.Getwd()
	if err != nil {
		fmt.Printf("err: %v\n", err)
	}
	fmt.Printf("dir: %v\n", dir)
}

// 获得临时目录
func getTemp() {
	s := os.TempDir()
	fmt.Printf("s: %v\n", s)
}

// 重命名文件
func renameFile() {
	err := os.Rename("test.txt", "test2.txt")
	if err != nil {
		fmt.Printf("err: %v\n", err)
	}
}

func main() {
	createDir()
	removeDir()
	getWd()
	getTemp()
	renameFile()
}

2. 文件操作

2.1 打开和关闭文件

// 关闭文件
func (f \*File) Close() error
// 以只读的方式打开文件
func Open(name string) (\*File, error)
// 以指定模式打开文件
func OpenFile(name string, flag int, perm FileMode) (\*File, error)

其中：

  • name：要打开的文件名
  • perm：文件权限，一个八进制数。r（读）04，w（写）02，x（执行）01。
  • flag：打开文件的模式。 模式有以下几种：

func main() {
	// 只读方式打开
	//file, err := os.Open("./test")

	file, err := os.OpenFile("./test.txt", os.O_RDWR | os.O_CREATE, 0777)
	if err != nil {
		fmt.Println("open file failed!, err:", err)
		return
	}
	// 关闭文件
	file.Close()
}

2.2 文件读取

file.Read()
func main() {
	file, err := os.Open("./test.txt")
	if err != nil {
		fmt.Println("open file failed!, err:", err)
		return
	}
	defer file.Close()

	// 循环读取文件
	var tmp = make([]byte, 128)
	for {
		n, err := file.Read(tmp)
		if err == io.EOF {	// 文件读取结束
			break
		}
		if err != nil {
			fmt.Println("read file failed, err:", err)
			return
		}
		fmt.Print(string(tmp[:n]))
	}
}
bufio
func main() {
	file, err := os.Open("./test.txt")
	if err != nil {
		fmt.Println("open file failed!, err:", err)
		return
	}
	defer file.Close()

	reader := bufio.NewReader(file)
	for {
		// 	line, err := reader.ReadBytes('\n')
		line, err := reader.ReadString('\n') //注意是字符
		if err == io.EOF {
			if len(line) != 0 {
				fmt.Println(line)
			}
			break
		}
		if err != nil {
			fmt.Println("read file failed, err:", err)
			return
		}
		fmt.Print(line)
	}
}
ioutil
func main() {
	content, err := ioutil.ReadFile("./test.txt")
	if err != nil {
		fmt.Println("read file failed, err:", err)
		return
	}
	fmt.Println(string(content))
}

2.3 文件写入

Write和WriteString
func main() {
	file, err := os.OpenFile("xx.txt", os.O_CREATE|os.O_TRUNC|os.O_WRONLY, 0666)
	if err != nil {
		fmt.Println("open file failed, err:", err)
		return
	}
	defer file.Close()
	str := "满腹经纶，白发不第；"
	file.Write([]byte(str))       //写入字节切片数据
	file.WriteString("才疏学浅，少年登科") //直接写入字符串数据
}
bufio
func main() {
	file, err := os.OpenFile("xx.txt", os.O_CREATE|os.O_TRUNC|os.O_WRONLY, 0666)
	if err != nil {
		fmt.Println("open file failed, err:", err)
		return
	}
	defer file.Close()
	writer := bufio.NewWriter(file)

	writer.Write([]byte("hello "))
	writer.WriteString("debuggingWorld")

	writer.Flush() //将缓存中的内容写入文件
}
ioutil.WriteFile
func main() {
	str := "hello debuggingWorld"
	err := ioutil.WriteFile("./xx.txt", []byte(str), 0666)
	if err != nil {
		fmt.Println("write file failed, err:", err)
		return
	}
}
