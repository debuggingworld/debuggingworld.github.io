from datasets import load_from_disk
from torch.utils.data import Dataset


class TrainModelDataset(Dataset):
    def __init__(self, split):
        # 从磁盘加载数据
        self.dataset = load_from_disk("./dataset/ChnSentiCorp")
        if split == "train":
            self.dataset = self.dataset["train"]
        elif split == "test":
            self.dataset = self.dataset["test"]
        elif split == "validation":
            self.dataset = self.dataset["validation"]
        else:
            print("数据名错误！")

    # 返回数据集长度
    def __len__(self):
        return len(self.dataset)

    # 对每条数据单独做处理
    def __getitem__(self, index):
        text = self.dataset[index]["text"]
        label = self.dataset[index]["label"]
        return text, label

1.2 模型选型/设计

增量微调 Bert 中文模型

import torch
from transformers import BertModel

# 定义设备信息
DEVICE = torch.device("cuda" if torch.cuda.is_available() else "cpu")
print(DEVICE)

# 加载预训练模型
modelPath = "./model/bert-base-chinese/models--bert-base-chinese/snapshots/c30a6ed22ab4564dc1e3b2ecbf6e766b0611a33f"
pretrained = BertModel.from_pretrained(modelPath).to(DEVICE)
print(pretrained)


# 定义下游任务（增量模型）
class MyModel(torch.nn.Module):
    def __init__(self):
        super().__init__()
        # 设计全连接网络，实现二分类任务
        self.fc = torch.nn.Linear(768, 2)

    # 使用模型处理数据（执行前向计算）
    def forward(self, input_ids, attention_mask, token_type_ids):
        # 全量训练
        # out = pretrained(input_ids=input_ids, attention_mask=attention_mask, token_type_ids=token_type_ids)
        # 冻结 Bert 模型的参数，让其不参与训练
        with torch.no_grad():
            out = pretrained(input_ids=input_ids, attention_mask=attention_mask, token_type_ids=token_type_ids)
        # 增量模型参与训练
        out = self.fc(out.last_hidden_state[:, 0])
        return out

# 模型训练
import torch
from torch.optim import AdamW
from torch.utils.data import DataLoader
from transformers import BertTokenizer

from debug_bert.net import MyModel
from debug_bert.trainModelDataset import TrainModelDataset

# 定义设备信息
DEVICE = torch.device("cuda" if torch.cuda.is_available() else "cpu")
# 定义训练的轮次(将整个数据集训练完一次为一轮)
EPOCH = 1000

# 加载字典和分词器
modelPath = "./model/bert-base-chinese/models--bert-base-chinese/snapshots/c30a6ed22ab4564dc1e3b2ecbf6e766b0611a33f"
tokenModel = BertTokenizer.from_pretrained(modelPath)


# 将传入的字符串进行编码
def collate_fn(data):
    text = [item[0] for item in data]
    label = [item[1] for item in data]
    # 编码
    data = tokenModel.batch_encode_plus(
        batch_text_or_text_pairs=text,
        add_special_tokens=True,
        # 当句子长度大于max_length(上限是model_max_length)时，截断
        truncation=True,
        max_length=512,
        # 一律补0到max_length
        padding="max_length",
        # 可取值为tf,pt,np,默认为list
        return_tensors="pt",
        # 返回序列长度
        return_length=True
    )
    input_ids = data["input_ids"]
    attention_mask = data["attention_mask"]
    token_type_ids = data["token_type_ids"]
    label = torch.LongTensor(label)
    return input_ids, attention_mask, token_type_ids, label


# 创建数据集
train_dataset = TrainModelDataset("train")
train_loader = DataLoader(
    dataset=train_dataset,
    # 训练批次
    batch_size=128,
    # 打乱数据集
    shuffle=True,
    # 舍弃最后一个批次的数据，防止形状出错
    drop_last=True,
    # 对加载的数据进行编码
    collate_fn=collate_fn
)

if __name__ == '__main__':
    # 开始训练
    print(DEVICE)
    model = MyModel().to(DEVICE)
    # 定义优化器
    optimizer = AdamW(model.parameters())
    # 定义损失函数
    loss_func = torch.nn.CrossEntropyLoss()

    for epoch in range(EPOCH):
        for i, (input_ids, attention_mask, token_type_ids, label) in enumerate(train_loader):
            # 将数据放到 DEVICE 上面
            input_ids, attention_mask, token_type_ids, label = (input_ids.to(DEVICE), attention_mask.to(DEVICE)
                                                                , token_type_ids.to(DEVICE), label.to(DEVICE))
            # 前向计算（将数据输入模型得到输出）
            out = model(input_ids, attention_mask, token_type_ids)
            # 根据输出计算损失
            loss = loss_func(out, label)
            # 根据误差优化参数
            optimizer.zero_grad()
            loss.backward()
            optimizer.step()

            # 每隔5个批次输出训练信息
            if i % 5 == 0:
                # 在类别维度（dim=1）取最大值索引，得到预测的类别标签
                out = out.argmax(dim=1)
                # 计算训练精度
                # (out == label) 生成布尔张量（如 [True, False]）
                # .sum()：统计 True 的数量
                # .item()：将张量转换为标量
                acc = (out == label).sum().item() / len(label)
                print(f"epoch:{epoch},i:{i},loss:{loss.item()},acc:{acc}")
        # 每训练完一轮，保存一次参数
        torch.save(model.state_dict(), f"params/{epoch}_bert.pth")
        print(epoch, "参数保存成功！")

# 模型使用接口（主观评估）
import torch
from transformers import BertTokenizer
from debug_bert.net import MyModel
from debug_bert.train import collate_fn

DEVICE = torch.device("cuda" if torch.cuda.is_available() else "cpu")
model = MyModel().to(DEVICE)
names = ["负向评价", "正向评价"]

modelPath = "./model/bert-base-chinese/models--bert-base-chinese/snapshots/c30a6ed22ab4564dc1e3b2ecbf6e766b0611a33f"
tokenModel = BertTokenizer.from_pretrained(modelPath)


def collate_fn(data):
    sentences = [data]
    data = tokenModel.batch_encode_plus(
        batch_text_or_text_pairs=sentences,
        truncation=True,
        max_length=512,
        padding="max_length",
        return_tensors="pt",
        return_length=True
    )
    input_ids = data["input_ids"]
    attention_mask = data["attention_mask"]
    token_type_ids = data["token_type_ids"]
    return input_ids, attention_mask, token_type_ids


def test():
    # 加载模型训练参数
    model.load_state_dict(torch.load("params/my_bert.pth", map_location=DEVICE))

    # 开启测试模型
    model.eval()

    while True:
        data = input("请输入测试数据（输入‘q’退出）：")
        if data == 'q':
            print("测试结束")
            break
        input_ids, attention_mask, token_type_ids = collate_fn(data)
        input_ids, attention_mask, token_type_ids = input_ids.to(DEVICE), attention_mask.to(DEVICE), token_type_ids.to(
            DEVICE)
        # 将数据输入到模型，得到输出
        with torch.no_grad():
            out = model(input_ids, attention_mask, token_type_ids)
            out = out.argmax(dim=1)
            print("模型判定：", names[out], "\n")


if __name__ == '__main__':
    test()

3. 优化

  1. 训练过程中进行评测，防止过拟合
  2. 只保存最优参数&最后一轮训练参数
import torch
from torch.optim import AdamW
from debug_bert.net import MyModel
from debug_bert.train import collate_fn
from debug_bert.trainModelDataset import TrainModelDataset
from torch.utils.data import DataLoader

DEVICE = torch.device("cuda" if torch.cuda.is_available() else "cpu")
# 定义训练的轮次(将整个数据集训练完一次为一轮)
EPOCH = 30000

# 创建训练数据集
train_dataset = TrainModelDataset("train")
train_loader = DataLoader(dataset=train_dataset, batch_size=50, shuffle=True, drop_last=True, collate_fn=collate_fn)
# 创建验证数据集 验证数据集没必要打乱
val_dataset = TrainModelDataset("validation")
val_loader = DataLoader(dataset=val_dataset, batch_size=50, shuffle=True, drop_last=False, collate_fn=collate_fn)

if __name__ == '__main__':
    print(DEVICE)
    model = MyModel().to(DEVICE)
    optimizer = AdamW(model.parameters())
    loss_func = torch.nn.CrossEntropyLoss()

    # 初始化验证最佳准确率
    best_val_acc = 0.0

    for epoch in range(EPOCH):
        model.train()
        for i, (input_ids, attention_mask, token_type_ids, label) in enumerate(train_loader):
            input_ids, attention_mask, token_type_ids, label = input_ids.to(DEVICE), attention_mask.to(
                DEVICE), token_type_ids.to(DEVICE), label.to(DEVICE)
            out = model(input_ids, attention_mask, token_type_ids)
            loss = loss_func(out, label)
            optimizer.zero_grad()
            loss.backward()
            optimizer.step()
            if i % 5 == 0:
                out = out.argmax(dim=1)
                acc = (out == label).sum().item() / len(label)
                print(f"epoch:{epoch},i:{i},loss:{loss.item()},acc:{acc}")
        # 验证模型（判断模型是否过拟合）
        # 设置为评估模型
        model.eval()
        # 不需要模型参与训练
        with torch.no_grad():
            val_acc = 0.0
            val_loss = 0.0
            for i, (input_ids, attention_mask, token_type_ids, label) in enumerate(val_loader):
                input_ids, attention_mask, token_type_ids, label = input_ids.to(DEVICE), attention_mask.to(
                    DEVICE), token_type_ids.to(DEVICE), label.to(DEVICE)
                out = model(input_ids, attention_mask, token_type_ids)
                # 根据输出计算损失
                val_loss += loss_func(out, label)
                # 根据数据，计算验证精度
                out = out.argmax(dim=1)
                val_acc += (out == label).sum().item()
            val_loss /= len(val_loader)
            val_acc /= len(val_loader)
            print(f"验证集：loss:{val_loss},acc:{val_acc}")
            # #每训练完一轮，保存一次参数
            # torch.save(model.state_dict(),f"params/{epoch}_bert.pth")
            # print(epoch,"参数保存成功！")
            # 根据验证准确率保存最优参数
            if val_acc > best_val_acc:
                best_val_acc = val_acc
                torch.save(model.state_dict(), "params1/best_bert.pth")
                print(f"EPOCH:{epoch}:保存最优参数：acc{best_val_acc}")
        # 保存最后一轮参数
        torch.save(model.state_dict(), "params1/last_bert.pth")
        print(f"EPOCH:{epoch}:最后一轮参数保存成功！")

1 训练模式（model.train()）

  • Dropout层：随机丢弃神经元，强制模型不依赖特定节点（正则化防止过拟合）。
  • BatchNorm层：基于当前训练 batch 的数据动态计算均值和方差，并更新全局统计量。
  • 其他层：例如部分自定义层可能有训练/推理阶段的不同行为。

2 评估模式（model.eval()）

  • Dropout层：停止随机丢弃，所有神经元激活（相当于关闭正则化）。
  • BatchNorm层：固定使用训练阶段积累的全局均值和方差，不再更新统计量。
  • 其他层：例如停止梯度跟踪，确保推理效率。
